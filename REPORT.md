### –ò–Ω—Ç–µ–≥—Ä–∞—Ü–∏—è —Å Pix2Pix (huggingface-pix2pix.js)

–ú–æ–¥—É–ª—å `huggingface-pix2pix.js` –æ–±–µ—Å–ø–µ—á–∏–≤–∞–µ—Ç –∫–æ–º–º—É–Ω–∏–∫–∞—Ü–∏—é –º–µ–∂–¥—É —Ñ—Ä–æ–Ω—Ç–µ–Ω–¥ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏–µ–º –∏ –ª–æ–∫–∞–ª—å–Ω—ã–º Python —Å–µ—Ä–≤–µ—Ä–æ–º Pix2Pix. –§—É–Ω–∫—Ü–∏—è `queryHuggingFacePix2Pix()` –æ—Ç–ø—Ä–∞–≤–ª—è–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –Ω–∞ —Å–µ—Ä–≤–µ—Ä –∏ –ø–æ–ª—É—á–∞–µ—Ç —Å–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ:

```javascript
const PROXY_API_URL = (import.meta.env.VITE_API_URL || 'http://localhost:3001') + '/api/pix2pix';

export async function queryHuggingFacePix2Pix(imageFile) {
  try {
    const resp = await fetch(PROXY_API_URL, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/octet-stream',
      },
      body: imageFile
    });

    if (!resp.ok) {
      const text = await resp.text();
      throw new Error(`Proxy error: ${text}`);
    }

    const contentType = resp.headers.get('content-type') || '';
    if (!contentType.startsWith('image/')) {
      throw new Error(`Expected image, got: ${contentType}`);
    }

    const blob = await resp.blob();
    return blob;
  } catch (error) {
    console.error('Query error:', error);
    throw error;
  }
}
```

–§—É–Ω–∫—Ü–∏—è –∏—Å–ø–æ–ª—å–∑—É–µ—Ç `application/octet-stream` –¥–ª—è –æ—Ç–ø—Ä–∞–≤–∫–∏ –±–∏–Ω–∞—Ä–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è. –û–Ω–∞ —Ç–∞–∫–∂–µ –ø—Ä–æ–≤–µ—Ä—è–µ—Ç —Ç–∏–ø –∫–æ–Ω—Ç–µ–Ω—Ç–∞ –æ—Ç–≤–µ—Ç–∞, —É–±–µ–∂–¥–∞—è—Å—å, —á—Ç–æ —Å–µ—Ä–≤–µ—Ä –≤–µ—Ä–Ω—É–ª –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ. –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω–æ –º–æ–¥—É–ª—å –ø—Ä–µ–¥–æ—Å—Ç–∞–≤–ª—è–µ—Ç —É—Ç–∏–ª–∏—Ç—ã `canvasToBlob()` –¥–ª—è –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è canvas –≤ PNG blob –∏ `revokeObjectUrl()` –¥–ª—è –æ—Å–≤–æ–±–æ–∂–¥–µ–Ω–∏—è –ø–∞–º—è—Ç–∏:

### Python Flask —Å–µ—Ä–≤–µ—Ä Pix2Pix (pix2pix_server.py)

–°–µ—Ä–¥—Ü–µ–º –±—ç–∫–µ–Ω–¥–∞ —è–≤–ª—è–µ—Ç—Å—è –∫–ª–∞—Å—Å `UNetGenerator`, –∫–æ—Ç–æ—Ä—ã–π —Ä–µ–∞–ª–∏–∑—É–µ—Ç –∞—Ä—Ö–∏—Ç–µ–∫—Ç—É—Ä—É U-Net –¥–ª—è —Ç—Ä–∞–Ω—Å—Ñ–æ—Ä–º–∞—Ü–∏–∏ —ç—Å–∫–∏–∑–æ–≤:

```python
class UNetGenerator(nn.Module):
    """U-Net –≥–µ–Ω–µ—Ä–∞—Ç–æ—Ä –¥–ª—è pix2pix"""
    def __init__(self):
        super().__init__()
        # Encoder - downsampling path
        self.e1 = nn.Sequential(nn.Conv2d(3, 64, 4, 2, 1), nn.LeakyReLU(0.2))
        self.e2 = nn.Sequential(nn.Conv2d(64, 128, 4, 2, 1), nn.BatchNorm2d(128), nn.LeakyReLU(0.2))
        self.e3 = nn.Sequential(nn.Conv2d(128, 256, 4, 2, 1), nn.BatchNorm2d(256), nn.LeakyReLU(0.2))
        # ... –µ—â—ë 5 —Å–ª–æ–µ–≤
        
        # Decoder - upsampling path with skip connections
        self.d1 = nn.Sequential(nn.ConvTranspose2d(512, 512, 4, 2, 1), nn.BatchNorm2d(512), nn.Dropout(0.5), nn.ReLU())
        self.d2 = nn.Sequential(nn.ConvTranspose2d(1024, 512, 4, 2, 1), nn.BatchNorm2d(512), nn.Dropout(0.5), nn.ReLU())
        # ... –µ—â—ë 6 —Å–ª–æ–µ–≤
    
    def forward(self, x):
        # Encoder
        e1 = self.e1(x)
        e2 = self.e2(e1)
        # ... —Å–æ—Ö—Ä–∞–Ω—è–µ–º –≤—Å–µ –ø—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω—ã–µ –≤—ã—Ö–æ–¥—ã
        e8 = self.e8(e7)
        
        # Decoder with skip connections
        d1 = self.d1(e8)
        d1 = torch.cat([d1, e7], 1)  # –ü—Ä–æ–ø—É—Å–∫–∞—é—â–µ–µ —Å–æ–µ–¥–∏–Ω–µ–Ω–∏–µ
        d2 = self.d2(d1)
        d2 = torch.cat([d2, e6], 1)  # –ü—Ä–æ–ø—É—Å–∫–∞—é—â–µ–µ —Å–æ–µ–¥–∏–Ω–µ–Ω–∏–µ
        # ... –ø—Ä–æ–¥–æ–ª–∂–∞–µ–º —Å–æ–µ–¥–∏–Ω–µ–Ω–∏—è –≤–≤–µ—Ä—Ö –ø–æ –∞—Ä—Ö–∏—Ç–µ–∫—Ç—É—Ä–µ
        
        return d8
```

–ö–æ–¥–∏—Ä–æ–≤—â–∏–∫ —Å–æ—Å—Ç–æ–∏—Ç –∏–∑ –≤–æ—Å—å–º–∏ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω—ã—Ö —Å–ª–æ–µ–≤ —Å–≤–µ—Ä—Ç–∫–∏ —Å LeakyReLU –∞–∫—Ç–∏–≤–∞—Ü–∏–µ–π, –∫–æ—Ç–æ—Ä—ã–µ –ø—Ä–æ–≥—Ä–µ—Å—Å–∏–≤–Ω–æ —Å–Ω–∏–∂–∞—é—Ç –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–æ–µ —Ä–∞–∑—Ä–µ—à–µ–Ω–∏–µ –ø—Ä–∏ —É–≤–µ–ª–∏—á–µ–Ω–∏–∏ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –∫–∞–Ω–∞–ª–æ–≤ (64 ‚Üí 128 ‚Üí 256 ‚Üí 512). –î–µ–∫–æ–¥–∏—Ä–æ–≤—â–∏–∫ –∏—Å–ø–æ–ª—å–∑—É–µ—Ç —Ç—Ä–∞–Ω—Å–ø–æ–Ω–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ —Å–≤–µ—Ä—Ç–∫–∏ (ConvTranspose2d) –¥–ª—è –≤–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏—è –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–æ–≥–æ —Ä–∞–∑—Ä–µ—à–µ–Ω–∏—è, —Å –ø—Ä–æ–ø—É—Å–∫–∞—é—â–∏–º–∏ —Å–æ–µ–¥–∏–Ω–µ–Ω–∏—è–º–∏, –∫–æ—Ç–æ—Ä—ã–µ –æ–±—ä–µ–¥–∏–Ω—è—é—Ç (concatenate) –≤—ã—Ö–æ–¥—ã –∏–∑ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏—Ö —Å–ª–æ–µ–≤ –∫–æ–¥–∏—Ä–æ–≤—â–∏–∫–∞. –≠—Ç–æ –ø–æ–∑–≤–æ–ª—è–µ—Ç —Å–æ—Ö—Ä–∞–Ω–∏—Ç—å –Ω–∏–∑–∫–æ—É—Ä–æ–≤–Ω–µ–≤—ã–µ –¥–µ—Ç–∞–ª–∏ –≤—Ö–æ–¥–Ω–æ–≥–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –≤ –≤—ã—Ö–æ–¥–Ω–æ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç–µ.

Flask –º–∞—Ä—à—Ä—É—Ç –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π —Ä–µ–∞–ª–∏–∑—É–µ—Ç –ø–æ–ª–Ω—ã–π –∫–æ–Ω–≤–µ–π–µ—Ä –æ–±—Ä–∞–±–æ—Ç–∫–∏:

```python
@app.route('/api/pix2pix', methods=['POST', 'OPTIONS'])
def pix2pix():
    """–ì–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –∏–∑ –≤—Ö–æ–¥–Ω–æ–≥–æ —ç—Å–∫–∏–∑–∞"""
    if request.method == 'OPTIONS':
        return '', 200
    
    try:
        initialize_model()
        
        # –ß–∏—Ç–∞–µ–º –≤—Ö–æ–¥–Ω–æ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
        image = Image.open(io.BytesIO(request.data)).convert('RGB')
        
        # –ò–∑–º–µ–Ω—è–µ–º —Ä–∞–∑–º–µ—Ä –µ—Å–ª–∏ –Ω—É–∂–Ω–æ
        if image.size != (256, 256):
            image = image.resize((256, 256), Image.LANCZOS)
        
        # –ù–æ—Ä–º–∞–ª–∏–∑—É–µ–º –≤ –¥–∏–∞–ø–∞–∑–æ–Ω [-1, 1]
        img_np = np.array(image, dtype=np.float32) / 127.5 - 1.0
        img_tensor = torch.from_numpy(img_np).permute(2, 0, 1).unsqueeze(0).to(device)
        
        # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –≤—ã—Ö–æ–¥–Ω–æ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
        with torch.no_grad():
            output = model(img_tensor)
        
        # –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ–º –æ–±—Ä–∞—Ç–Ω–æ –≤ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ ([-1, 1] -> [0, 255])
        output_np = (output[0].permute(1, 2, 0).cpu().numpy() + 1) / 2 * 255
        output_np = np.clip(output_np, 0, 255).astype(np.uint8)
        output_img = Image.fromarray(output_np)
        
        # –í–æ–∑–≤—Ä–∞—â–∞–µ–º PNG
        img_io = io.BytesIO()
        output_img.save(img_io, 'PNG')
        img_io.seek(0)
        
        return send_file(img_io, mimetype='image/png')
        
    except Exception as e:
        logger.error(f"‚ùå –û—à–∏–±–∫–∞: {e}")
```
